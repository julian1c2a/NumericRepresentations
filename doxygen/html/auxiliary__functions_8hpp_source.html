50// precondition: low*low <= n, high*high > n.
51constexpr inline
52size_t ceilsqrt(size_t n,size_t low,size_t high) noexcept
53{
54 const auto mid{std::midpoint(low, high)};
55 const auto sqmid{mid*mid};
56 return
57 (
58 (low + 1 >= high) ?
59 high :
60 (sqmid == n) ?
61 mid :
62 (sqmid < n) ?
63 ceilsqrt(n, mid, high) :
64 ceilsqrt(n, low, mid)
65 );
66}
